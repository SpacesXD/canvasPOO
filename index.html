API
Métodos que canvasPOO agrega al ctx.
ctx.circle(x, y, r)forma

Círculo relleno. Usa el fillStyle actual.

paramtipodesc
xnumberCentro X
ynumberCentro Y
rnumberRadio
ctx.fillStyle = 'red'
ctx.circle(200, 200, 40)
ctx.strokeCircle(x, y, r)forma

Borde de círculo. Usa strokeStyle y lineWidth actuales.

ctx.strokeStyle = 'cyan'
ctx.lineWidth = 2

ctx.roundRect(x, y, w, h, r?)forma

Rectángulo con esquinas redondeadas. r default: 8.

ctx.line(x1, y1, x2, y2)forma

Línea directa entre dos puntos.

ctx.polygon(points)forma

Polígono cerrado desde array de [x,y].

Colisión entre dos Sprites (o cualquier objeto con size). Usa pivote central.

if (ctx.collidesSprite(player, enemy)) {

}

Colisión entre dos Rect. Los objetos usan su posición como centro.

Colisión entre dos Circle.

ctx.loop(fn)loop

Arranca el game loop. fn recibe dt (segundos desde el último frame) y time (tiempo total). Devuelve una función para pararlo.

const caja = ctx.Rect(100,100,60,60,'#4db8ff')

const stop = ctx.loop((dt, time) => {
  ctx.background('#111')
  caja.rotation += dt
  caja.draw()
})

// para pararlo:
stop()
ctx.Camera(opts?)objeto

Crea una cámara. Asígnala a ctx.camera para activarla. Soporta follow, zoom y suavizado.

const cam = ctx.Camera({ zoom: 1.5, smoothing: 0.1 });
cam.follow(player);
ctx.camera = cam;

ctx.loop((dt) => {
  cam.update(dt);
  player.draw();
  cam.screen(() => { score.draw(); }); // UI fija
});
ctx.Scene()objeto

Contenedor opcional de objetos. Maneja update, draw y limpia los destruidos automáticamente.

const scene = ctx.Scene();
scene.add(player, enemy, background);

ctx.loop((dt) => {
  ctx.background('#111');
  scene.tick(dt); // update + draw
});

// buscar por tag
scene.getByTag('enemy');

ctx.tween(obj, props, duration, easing?, onDone?)utilidad

Anima propiedades numéricas de cualquier objeto. El loop lo actualiza automático.

ctx.tween(player, { x: 400 }, 1.5);

// con easing y callback:
ctx.tween(enemy, { alpha: 0 }, 0.5, t => t * t, () => {
  enemy.destroy();
});
ctx.shake(intensity, duration)utilidad

Sacude el canvas. Se integra solo en el loop.

ctx.shake(10, 0.3); // intensidad 10px, 0.3 segundos
ctx.gravity(v) / ctx.friction(v)física

Define gravedad y fricción globales. Todos los objetos con física las heredan, salvo que definan la suya.

ctx.gravity(500);
ctx.friction(0.02);

const ball = ctx.Circle(200, 0, 20, 'red');
ball.enablePhysics(); // usa la global
ball.bounce = true;   // rebota en bordes
// ball.wrap = true;  // o aparece por el otro lado

function spawn(x, y) {
  const p = ctx.Circle(x, y, Math.random()*6+2, '#4db8ff')
  p.vx = (Math.random()-.5)*200
  p.vy = (Math.random()-.5)*200
  p.vida = 1
  return p
}

PASO 3

Tubos

const GAP = 160, PIPE_W = 52, SPEED = 180;
let pipes = [];

function spawnPipe() {
  const top = ctx.randInt(60, canvas.height - GAP - 60);
  pipes.push({
    top:   ctx.Rect(canvas.width + PIPE_W/2, top/2,         PIPE_W, top,                '#4caf50'),
    bot:   ctx.Rect(canvas.width + PIPE_W/2, top+GAP + (canvas.height-top-GAP)/2, PIPE_W, canvas.height-top-GAP, '#4caf50'),
    passed: false,
  });
}

function updatePipes(dt) {
  for (const p of pipes) {
    p.top.position.x -= SPEED * dt;
    p.bot.position.x -= SPEED * dt;
  }
  pipes = pipes.filter(p => p.top.position.x > -PIPE_W);
}
PASO 4

Colisiones y contador

let score = 0;
const scoreText = ctx.print('0', canvas.width/2, 40, {
  size: 48, bold: true, color: '#fff', align: 'center',
});

function checkCollisions() {
  if (bird.position.y + bird.radius > canvas.height || bird.position.y - bird.radius < 0) {
    state = 'dead';
  }
  for (const p of pipes) {
    if (ctx.collidesCircleRect(bird, p.top) || ctx.collidesCircleRect(bird, p.bot)) {
      state = 'dead';
    }
    if (!p.passed && p.top.position.x < bird.position.x) {
      p.passed = true;
      score++;
      scoreText.text = String(score);
    }
  }
}
PASO 5

Loop principal

let state = 'playing';
let pipeTimer = 0;

function init() {
  bird.position.y = 250;
  bird.velocity.y = 0;
  pipes = [];
  score = 0;
  scoreText.text = '0';
  state = 'playing';
  pipeTimer = 0;
}

ctx.loop((dt) => {
  ctx.background('#70c5ce');

  if (state === 'playing') {
    bird.update(dt);
    pipeTimer += dt;
    if (pipeTimer > 1.6) { spawnPipe(); pipeTimer = 0; }
    updatePipes(dt);
    checkCollisions();
  }

  for (const p of pipes) { p.top.draw(); p.bot.draw(); }
  bird.draw();
  scoreText.draw();

  if (state === 'dead') {
    ctx.withAlpha(0.5, () => ctx.background('#000'));
    ctx.print('GAME OVER', canvas.width/2, canvas.height/2 - 20, {
      size: 36, bold: true, color: '#fff', align: 'center',
    }).draw();
    ctx.print('toca para reiniciar', canvas.width/2, canvas.height/2 + 30, {
      size: 16, color: '#ccc', align: 'center',
    }).draw();
  }
});

init();
